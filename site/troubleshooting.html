Troubleshooting

Fast diagnosis commands

Essential checks

mosaic --project-state status
mosaic --project-state health
mosaic --project-state doctor

404 or model list failed

# check configured URL and key env name
mosaic --project-state configure --show

# Azure base URL should be:
# https://.openai.azure.com/openai/v1
# avoid duplicated /v1 in URL value

Command not found: mosaic

# zsh
export PATH="$HOME/.cargo/bin:$PATH"
source ~/.zshrc

# verify
which mosaic
mosaic --version

Channel/gateway operations

mosaic --project-state channels list
mosaic --project-state channels logs --channel all --tail 50
mosaic --project-state gateway probe
mosaic --project-state logs --tail 80
